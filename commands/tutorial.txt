Darknet to ONNX

python main.py \
--mode darknet \
--cfg input/yolov3-tiny.cfg \
--weights input/yolov3-tiny.weights \
--output yolov3-tiny.onnx \
--shape 1 3 416 416 \
--opset 18 \
--validate \
--no-simplify

Darknet to PyTorch

python main.py \
--mode darknet \
--cfg yolov4.cfg \
--weights yolov4.weights \
--output_mode pt \
--output yolov4_torch.pt

PyTorch to ONNX

python main.py \
--mode pytorch \
--weights yolov8n.pt \
--output yolov8n_npu.onnx \
--shape 1 3 416 416 \
--validate
